<template>
    <f7-page>
        <f7-navbar>
            <f7-nav-left>
                <f7-link back text="返回"></f7-link>
            </f7-nav-left>
            <div class="title">图片预览测试</div>
        </f7-navbar>
        <div style="margin: 10px">
            <div v-for="(item,idx) in images"
                :key="idx"
                :style="computedStyle(item)"
                class="image_view_item"
                >
            </div>
        </div>
        <button @click="uploadImages">点击上传</button>
    </f7-page>
</template>
<script>
import {readFileAsImageAllPlatform, uploadFilePromise} from '../components/upload_comp/util.js'
import map from 'lodash/map'
export default {
    
    data() {
        return {
            images: [],
        }
    },
 
    methods: {
        computedStyle: function(item) {
            console.log('jjjff', item);
            return `background-image:url(${item});`
        },
        uploadImages: async function() {
            let files = this.$f7Route.context.files;  
            console.log(files);
             for (let i = 0; i < files.length; i++) {
                const file = files[i];
                let resp = await uploadFilePromise(file, this.uploadProgress);
                console.log(resp);
                if (!resp.err) {
                    this.$f7.dialog.alert("上传成功");
                }
            }
        },
        uploadProgress: function(progress) {
            console.log('upload progress:', progress)
        }
    },
 
    created() {
        let files = this.$f7Route.context.files;   
        console.log(files);
        Promise.all(map(files, file => readFileAsImageAllPlatform(file)))
            .then(images => {
                console.log('images:', images);
                this.images = images;
            })
    }
 
}
</script>
